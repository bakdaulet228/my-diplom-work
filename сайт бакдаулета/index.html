Проектирование архитектуры

Разработка микросервисной архитектуры и проектирование БД

Микросервисная архитектура

Система разделена на независимые микросервисы, каждый из которых отвечает за определенную бизнес-логику:

  • Auth Service - аутентификация и авторизация
  • Project Service - управление проектами
  • Task Service - управление задачами
  • Notification Service - система уведомлений
  • API Gateway - единая точка входа

Пример конфигурации Spring Cloud Gateway

application.yml
spring:
  cloud:
    gateway:
      routes:
        - id: auth-service
          uri: lb://auth-service
          predicates:
            - Path=/api/auth/**
          filters:
            - StripPrefix=1
            
        - id: project-service
          uri: lb://project-service
          predicates:
            - Path=/api/projects/**
          filters:
            - StripPrefix=1
            
        - id: task-service
          uri: lb://task-service
          predicates:
            - Path=/api/tasks/**
          filters:
            - StripPrefix=1

eureka:
  client:
    service-url:

@Entity
@Table(name = "projects")
@Data
@NoArgsConstructor
@AllArgsConstructor
public class Project {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    
    @Column(nullable = false)
    private String name;
    
    private String description;
    
    @Enumerated(EnumType.STRING)
    private ProjectStatus status;
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "owner_id")
    private User owner;
    
    @OneToMany(mappedBy = "project", cascade = CascadeType.ALL)
    private Set tasks = new HashSet<>();
    
    @CreationTimestamp
    private LocalDateTime createdAt;
    
    @UpdateTimestamp
    private LocalDateTime updatedAt;
}

Пример unit-теста

@ExtendWith(MockitoExtension.class)
class ProjectServiceTest {

    @Mock
    private ProjectRepository projectRepository;
    
    @Mock
    private UserService userService;
    
    @InjectMocks
    private ProjectService projectService;

    @Test
    void createProject_ShouldReturnCreatedProject() {
        // Given
        User owner = new User(1L, "john@example.com", "John Doe");
        Project project = new Project();
        project.setName("Test Project");
        
        when(userService.findById(1L)).thenReturn(Optional.of(owner));
        when(projectRepository.save(any(Project.class))).thenReturn(project);

        // When
        Project result = projectService.createProject(project, 1L);

        // Then
        assertNotNull(result);
        assertEquals("Test Project", result.getName());
        assertEquals(owner, result.getOwner());
        verify(projectRepository, times(1)).save(project);
    }
}
